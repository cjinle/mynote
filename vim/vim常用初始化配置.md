# vim常用初始化配置

```bash
let &termencoding=&encoding
set fileencodings=utf-8,gbk,gb2312,ucs-bom,cp936
set nu
set autoindent
set tabstop=4
set softtabstop=4
set shiftwidth=4
set expandtab
set smarttab
set nobackup
set nowrap

set backup
set backupdir=C:\WINDOWS\Temp
set backupskip=C:\WINDOWS\Temp\*
set directory=C:\WINDOWS\Temp
set writebackup

# 设置文件格式
set ff=unix
set ff=dos
```
